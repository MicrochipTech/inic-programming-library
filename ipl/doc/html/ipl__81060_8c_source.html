41 uint8_t OS81060_ProgConfigString(uint32_t lData, uint8_t pData[])
42 {
43  uint8_t res;
44  uint32_t loop;
45  Ipl_Trace(IPL_TRACETAG_INFO, "OS81060_ProgConfigString called");
46  /* Get addresses and sizes from metadata */
47  res = Ipl_ParseIpf(&Ipl_IpfData, lData, pData, STRINGTYPE_META);
48  if (IPL_RES_OK == res)
49  {
50  /* Check if IPF contains proper ConfigString */
51  res = Ipl_ParseIpf(&Ipl_IpfData, lData, pData, STRINGTYPE_CONFIG);
52  if (IPL_RES_OK == res)
53  {
54  /* Check if IPF fits to INIC */
55  res = Ipl_CheckChipId();
56  if (IPL_RES_OK == res)
57  {
58  /* Write CS */
59  Ipl_IplData.Tel[0] = CMD_LEG_WRITECS;
60  Ipl_IplData.Tel[1] = 0x00U;
61  Ipl_IplData.Tel[2] = 0x00U;
62  Ipl_IplData.Tel[3] = 0x20U;
63  for(loop=0U; loop<32U; loop++)
64  {
65  Ipl_IplData.Tel[4U+loop] = Ipl_PData(loop+Ipl_IpfData.StringOffset, lData, pData);
66  }
67  Ipl_IplData.TelLen = 36U;
68  res = Ipl_ExecInicCmd();
69  }
70  }
71  }
72  Ipl_Trace(Ipl_TraceTag(res), "OS81060_ProgConfigString returned 0x%02X", res);
73  return res;
74 }

78 uint8_t OS81060_ReadConfigStringVersion(uint32_t lData, uint8_t pData[])
79 {
80  uint8_t res;
81  Ipl_Trace(IPL_TRACETAG_INFO, "OS81060_ReadConfigStringVersion called");
82  /* Get CS Version */
83  Ipl_ClrTel();
84  Ipl_IplData.Tel[0] = CMD_LEG_GETCSINFO;
85  Ipl_IplData.Tel[3] = 0x20U;
86  Ipl_IplData.TelLen = CMD_LEG_GETCSINFO_TXLEN;
87  res = Ipl_ExecInicCmd();
88  if (IPL_RES_OK == res)
89  {
90  Ipl_InicData.CfgsVersionValid = VERSION_VALID;
91  Ipl_InicData.CfgsCustMajorVersion = Ipl_IplData.Tel[6];
92  Ipl_InicData.CfgsCustMinorVersion = Ipl_IplData.Tel[7];
93  Ipl_InicData.CfgsCustReleaseVersion = Ipl_IplData.Tel[8];
94  Ipl_Trace(Ipl_TraceTag(res), "OS81060_ReadConfigStringVersion returned 0x%02X - V%u.%u.%u", res,
95  Ipl_InicData.CfgsCustMajorVersion, Ipl_InicData.CfgsCustMinorVersion, Ipl_InicData.CfgsCustReleaseVersion);
96  }
97  else
98  {
99  Ipl_Trace(Ipl_TraceTag(res), "OS81060_ReadConfigStringVersion returned 0x%02X", res);
100  }
101  return res;
102 }
