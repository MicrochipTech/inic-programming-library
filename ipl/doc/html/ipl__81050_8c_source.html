41 uint8_t OS81050_ProgFirmware(uint32_t lData, uint8_t pData[])
42 {
43  uint8_t res;
44  uint32_t loop, adr, startadr, stopadr;
45 
46  Ipl_Trace(IPL_TRACETAG_INFO, "OS81050_ProgFirmware called");
47  /* Get addresses and sizes from metadata */
48  res = Ipl_ParseIpf(&Ipl_IpfData, lData, pData, STRINGTYPE_META);
49  if (IPL_RES_OK == res)
50  {
51  /* Check if IPF contains proper Firmware */
52  res = Ipl_ParseIpf(&Ipl_IpfData, lData, pData, STRINGTYPE_FW);
53  if (IPL_RES_OK == res)
54  {
55  /* Check if IPF fits to INIC */
56  res = Ipl_CheckChipId();
57  if (IPL_RES_OK == res)
58  {
59  /* Erase INIC FW */
60  Ipl_ClrTel();
61  Ipl_IplData.Tel[0] = CMD_ERASEPROGMEM;
62  Ipl_IplData.Tel[1] = 0x08U;
63  Ipl_IplData.Tel[2] = 0x77U;
64  Ipl_IplData.TelLen = CMD_ERASEPROGMEM_TXLEN;
65  res = Ipl_ExecInicCmd();
66  if (IPL_RES_OK == res)
67  {
68  /* Select Flash Page 0 */
69  Ipl_ClrTel();
70  Ipl_IplData.Tel[0] = CMD_SETPROGMEMPAGE;
71  Ipl_IplData.Tel[1] = 0x0U;
72  Ipl_IplData.TelLen = CMD_SETPROGMEMPAGE_TXLEN;
73  res = Ipl_ExecInicCmd();
74  if (IPL_RES_OK == res)
75  {
76  /* Clear CRC */
77  Ipl_ClrTel();
78  Ipl_IplData.Tel[0] = CMD_CLEARCRC;
79  Ipl_IplData.TelLen = CMD_CLEARCRC_TXLEN;
80  res = Ipl_ExecInicCmd();
81  if (IPL_RES_OK == res)
82  {
83  /* Write FW Page 0 */
84  startadr = 0x2000U;
85  adr = startadr;
86  stopadr = 0x10000U;
87  while ( (adr < stopadr) && (IPL_RES_OK == res) )
88  {
89  Ipl_IplData.Tel[0] = CMD_WRITEPROGMEM;
90  Ipl_IplData.Tel[1] = (adr >> 8U) & 0xFFU;
91  Ipl_IplData.Tel[2] = adr & 0xFFU;
92  Ipl_IplData.Tel[3] = 0x20U;
93  for(loop=0U; loop<32U; loop++)
94  {
95  Ipl_IplData.Tel[4U+loop] = Ipl_PData((loop+Ipl_IpfData.StringOffset+adr)-startadr, lData, pData);
96  }
97  Ipl_IplData.TelLen = 36U;
98  res = Ipl_ExecInicCmd();
99  adr += 0x20U;
100  Ipl_ProgressIndicator(adr-startadr, Ipl_IpfData.StringSize);
101  }
102  if (IPL_RES_OK == res)
103  {
104  /* Select Flash Page 1 */
105  Ipl_ClrTel();
106  Ipl_IplData.Tel[0] = CMD_SETPROGMEMPAGE;
107  Ipl_IplData.Tel[1] = 0x1U;
108  Ipl_IplData.TelLen = CMD_SETPROGMEMPAGE_TXLEN;
109  res = Ipl_ExecInicCmd();
110  if (IPL_RES_OK == res)
111  {
112  /* Write FW Page 1 */
113  adr = 0x10000U;
114  stopadr = 0x1FBFFU;
115  while ( (adr < stopadr) && (IPL_RES_OK == res) )
116  {
117  Ipl_IplData.Tel[0] = CMD_WRITEPROGMEM;
118  Ipl_IplData.Tel[1] = (adr >> 8U) & 0xFFU;
119  Ipl_IplData.Tel[2] = adr & 0xFFU;
120  Ipl_IplData.Tel[3] = 0x20U;
121  for(loop=0U; loop<32U; loop++)
122  {
123  Ipl_IplData.Tel[4U+loop] = Ipl_PData((loop+Ipl_IpfData.StringOffset+adr)-startadr, lData, pData);
124  }
125  Ipl_IplData.TelLen = 36U;
126  res = Ipl_ExecInicCmd();
127  adr += 0x20U;
128  Ipl_ProgressIndicator(adr-startadr, Ipl_IpfData.StringSize);
129  }
130  if (IPL_RES_OK == res)
131  {
132  /* Get CRC */
133  Ipl_ClrTel();
134  Ipl_IplData.Tel[0] = CMD_GETCRC;
135  Ipl_IplData.Tel[3] = 0x2U;
136  Ipl_IplData.TelLen = CMD_GETCRC_TXLEN;
137  res = Ipl_ExecInicCmd();
138  if (IPL_RES_OK == res)
139  {
140  if ( (Ipl_IplData.Tel[4] != 0x00U) || (Ipl_IplData.Tel[5] != 0x00U) )
141  {
142  res = IPL_RES_ERR_WRONG_CRC;
143  }
144  }
145  }
146  }
147  }
148  }
149  }
150  }
151  }
152  }
153  }
154  Ipl_ProgressIndicator(1U, 1U); /* Set Progress Indicator to 100 */
155  Ipl_Trace(Ipl_TraceTag(res), "OS81050_ProgFirmware returned 0x%02X", res);
156  return res;
157 }

162 uint8_t OS81050_ProgConfigString(uint32_t lData, uint8_t pData[])
163 {
164  uint8_t res;
165  uint32_t loop, page;
166  Ipl_Trace(IPL_TRACETAG_INFO, "OS81050_ProgConfigString called");
167  /* Get addresses and sizes from metadata */
168  res = Ipl_ParseIpf(&Ipl_IpfData, lData, pData, STRINGTYPE_META);
169  if (IPL_RES_OK == res)
170  {
171  /* Check if IPF contains proper ConfigString */
172  res = Ipl_ParseIpf(&Ipl_IpfData, lData, pData, STRINGTYPE_CONFIG);
173  if (IPL_RES_OK == res)
174  {
175  /* Check if IPF fits to INIC */
176  res = Ipl_CheckChipId();
177  if (IPL_RES_OK == res)
178  {
179  /* Erase INIC CS */
180  Ipl_ClrTel();
181  Ipl_IplData.Tel[0] = CMD_LEG_ERASECS;
182  Ipl_IplData.Tel[1] = 0x67U;
183  Ipl_IplData.TelLen = CMD_LEG_ERASECS_TXLEN;
184  res = Ipl_ExecInicCmd();
185  if (IPL_RES_OK == res)
186  {
187  /* Clear CRC */
188  Ipl_ClrTel();
189  Ipl_IplData.Tel[0] = CMD_CLEARCRC;
190  Ipl_IplData.TelLen = CMD_CLEARCRC_TXLEN;
191  res = Ipl_ExecInicCmd();
192  if (IPL_RES_OK == res)
193  {
194  /* Write CS */
195  for(page=0U; page<=3U; page++)
196  {
197  Ipl_IplData.Tel[0] = CMD_LEG_WRITECS;
198  Ipl_IplData.Tel[1] = 0x00U;
199  Ipl_IplData.Tel[2] = (uint8_t) page*32U;
200  Ipl_IplData.Tel[3] = 0x20U;
201  for(loop=0U; loop<32U; loop++)
202  {
203  Ipl_IplData.Tel[4U+loop] = Ipl_PData(loop+Ipl_IpfData.StringOffset+(uint32_t) (page*32U), lData, pData);
204  }
205  Ipl_IplData.TelLen = 36U;
206  res = Ipl_ExecInicCmd();
207  Ipl_ProgressIndicator((page*32U), Ipl_IpfData.StringSize);
208  }
209  if (IPL_RES_OK == res)
210  {
211  /* Get CRC */
212  Ipl_ClrTel();
213  Ipl_IplData.Tel[0] = CMD_GETCRC;
214  Ipl_IplData.Tel[3] = 0x2U;
215  Ipl_IplData.TelLen = CMD_GETCRC_TXLEN;
216  res = Ipl_ExecInicCmd();
217  if(IPL_RES_OK == res)
218  {
219  if ( (Ipl_IplData.Tel[4] != 0x00U) || (Ipl_IplData.Tel[5] != 0x00U) )
220  {
221  res = IPL_RES_ERR_WRONG_CRC;
222  }
223  }
224  }
225  }
226  }
227  }
228  }
229  }
230  Ipl_ProgressIndicator(1U, 1U); /* Set Progress Indicator to 100 */
231  Ipl_Trace(Ipl_TraceTag(res), "OS81050_ProgConfigString returned 0x%02X", res);
232  return res;
233 }
