41 uint8_t OS81110_ProgFirmware(uint32_t lData, uint8_t pData[])
42 {
43  uint8_t res;
44  uint32_t loop, adr, startadr, stopadr;
45 
46  Ipl_Trace(IPL_TRACETAG_INFO, "OS81110_ProgFirmware called");
47  /* Get addresses and sizes from metadata */
48  res = Ipl_ParseIpf(&Ipl_IpfData, lData, pData, STRINGTYPE_META);
49  if (IPL_RES_OK == res)
50  {
51  /* Check if IPF contains proper Firmware */
52  res = Ipl_ParseIpf(&Ipl_IpfData, lData, pData, STRINGTYPE_FW);
53  if (IPL_RES_OK == res)
54  {
55  /* Check if IPF fits to INIC */
56  res = Ipl_CheckChipId();
57  if (IPL_RES_OK == res)
58  {
59  /* Erase Enable */
60  Ipl_ClrTel();
61  Ipl_IplData.Tel[0] = CMD_LEG_ERASEENABLE;
62  Ipl_IplData.Tel[2] = 0x07U;
63  Ipl_IplData.TelLen = CMD_LEG_ERASEENABLE_TXLEN;
64  res = Ipl_ExecInicCmd();
65  if (IPL_RES_OK == res)
66  {
67  /* Erase INIC FW */
68  Ipl_ClrTel();
69  Ipl_IplData.Tel[0] = CMD_ERASEPROGMEM;
70  Ipl_IplData.Tel[1] = 0x06U;
71  Ipl_IplData.Tel[2] = 0x7AU;
72  Ipl_IplData.TelLen = CMD_ERASEPROGMEM_TXLEN;
73  res = Ipl_ExecInicCmd();
74  if (IPL_RES_OK == res)
75  {
76  /* Select Flash Page 0 */
77  Ipl_ClrTel();
78  Ipl_IplData.Tel[0] = CMD_SETPROGMEMPAGE;
79  Ipl_IplData.Tel[1] = 0x0U;
80  Ipl_IplData.TelLen = CMD_SETPROGMEMPAGE_TXLEN;
81  res = Ipl_ExecInicCmd();
82  if (IPL_RES_OK == res)
83  {
84  /* Clear CRC */
85  Ipl_ClrTel();
86  Ipl_IplData.Tel[0] = CMD_CLEARCRC;
87  Ipl_IplData.TelLen = CMD_CLEARCRC_TXLEN;
88  res = Ipl_ExecInicCmd();
89  if (IPL_RES_OK == res)
90  {
91  /* Write FW Page 0 */
92  startadr = 0x1800U;
93  adr = startadr;
94  stopadr = 0x10000U;
95  while ( (adr < stopadr) && (IPL_RES_OK == res) )
96  {
97  Ipl_IplData.Tel[0] = CMD_WRITEPROGMEM;
98  Ipl_IplData.Tel[1] = (adr >> 8U) & 0xFFU;
99  Ipl_IplData.Tel[2] = adr & 0xFFU;
100  Ipl_IplData.Tel[3] = 0x20U;
101  for(loop=0U; loop<32U; loop++)
102  {
103  Ipl_IplData.Tel[4U+loop] = Ipl_PData((loop+Ipl_IpfData.StringOffset+adr)-startadr, lData, pData);
104  }
105  Ipl_IplData.TelLen = 36U;
106  res = Ipl_ExecInicCmd();
107  adr += 0x20U;
108  Ipl_ProgressIndicator(adr-startadr, Ipl_IpfData.StringSize);
109  }
110  if (IPL_RES_OK == res)
111  {
112  /* Select Flash Page 1 */
113  Ipl_ClrTel();
114  Ipl_IplData.Tel[0] = CMD_SETPROGMEMPAGE;
115  Ipl_IplData.Tel[1] = 0x1U;
116  Ipl_IplData.TelLen = CMD_SETPROGMEMPAGE_TXLEN;
117  res = Ipl_ExecInicCmd();
118  if (IPL_RES_OK == res)
119  {
120  /* Write FW Page 1 */
121  adr = 0x10000U;
122  stopadr = 0x1FFFFU;
123  while ( (adr < stopadr) && (IPL_RES_OK == res) )
124  {
125  Ipl_IplData.Tel[0] = CMD_WRITEPROGMEM;
126  Ipl_IplData.Tel[1] = (adr >> 8U) & 0xFFU;
127  Ipl_IplData.Tel[2] = adr & 0xFFU;
128  Ipl_IplData.Tel[3] = 0x20U;
129  for(loop=0U; loop<32U; loop++)
130  {
131  Ipl_IplData.Tel[4U+loop] = Ipl_PData((loop+Ipl_IpfData.StringOffset+adr)-startadr, lData, pData);
132  }
133  Ipl_IplData.TelLen = 36U;
134  res = Ipl_ExecInicCmd();
135  adr += 0x20U;
136  Ipl_ProgressIndicator(adr-startadr, Ipl_IpfData.StringSize);
137  }
138  if (IPL_RES_OK == res)
139  {
140  /* Get CRC */
141  Ipl_ClrTel();
142  Ipl_IplData.Tel[0] = CMD_GETCRC;
143  Ipl_IplData.Tel[3] = 0x2U;
144  Ipl_IplData.TelLen = CMD_GETCRC_TXLEN;
145  res = Ipl_ExecInicCmd();
146  if (IPL_RES_OK == res)
147  {
148  if ( (Ipl_IplData.Tel[4] != 0x00U) || (Ipl_IplData.Tel[5] != 0x00U) )
149  {
150  res = IPL_RES_ERR_WRONG_CRC;
151  }
152  }
153  }
154  }
155  }
156  }
157  }
158  }
159  }
160  }
161  }
162  }
163  Ipl_ProgressIndicator(1U, 1U); /* Set Progress Indicator to 100 */
164  Ipl_Trace(Ipl_TraceTag(res), "OS81110_ProgFirmware returned 0x%02X", res);
165  return res;
166 }

170 uint8_t OS81110_ProgConfigString(uint32_t lData, uint8_t pData[])
171 {
172  uint8_t res;
173  uint32_t loop, page;
174  Ipl_Trace(IPL_TRACETAG_INFO, "OS81110_ProgConfigString called");
175  /* Get addresses and sizes from metadata */
176  res = Ipl_ParseIpf(&Ipl_IpfData, lData, pData, STRINGTYPE_META);
177  if (IPL_RES_OK == res)
178  {
179  /* Check if IPF contains proper ConfigString */
180  res = Ipl_ParseIpf(&Ipl_IpfData, lData, pData, STRINGTYPE_CONFIG);
181  if (IPL_RES_OK == res)
182  {
183  /* Check if IPF fits to INIC */
184  res = Ipl_CheckChipId();
185  if (IPL_RES_OK == res)
186  {
187  /* Erase Enable */
188  Ipl_ClrTel();
189  Ipl_IplData.Tel[0] = CMD_LEG_ERASEENABLE;
190  Ipl_IplData.Tel[2] = 0x07U;
191  Ipl_IplData.TelLen = CMD_LEG_ERASEENABLE_TXLEN;
192  res = Ipl_ExecInicCmd();
193  if (IPL_RES_OK == res)
194  {
195  /* Erase INIC CS */
196  Ipl_ClrTel();
197  Ipl_IplData.Tel[0] = CMD_LEG_ERASECS;
198  Ipl_IplData.Tel[1] = 0x67U;
199  Ipl_IplData.TelLen = CMD_LEG_ERASECS_TXLEN;
200  res = Ipl_ExecInicCmd();
201  if (IPL_RES_OK == res)
202  {
203  /* Clear CRC */
204  Ipl_ClrTel();
205  Ipl_IplData.Tel[0] = CMD_CLEARCRC;
206  Ipl_IplData.TelLen = CMD_CLEARCRC_TXLEN;
207  res = Ipl_ExecInicCmd();
208  if (IPL_RES_OK == res)
209  {
210  /* Write CS */
211  for(page=0U; page<=3U; page++)
212  {
213  Ipl_IplData.Tel[0] = CMD_LEG_WRITECS;
214  Ipl_IplData.Tel[1] = 0x00U;
215  Ipl_IplData.Tel[2] = (uint8_t) page*32U;
216  Ipl_IplData.Tel[3] = 0x20U;
217  for(loop=0U; loop<32U; loop++)
218  {
219  Ipl_IplData.Tel[4U+loop] = Ipl_PData(loop+Ipl_IpfData.StringOffset+(uint32_t) (page*32U), lData, pData);
220  }
221  Ipl_IplData.TelLen = 36U;
222  res = Ipl_ExecInicCmd();
223  Ipl_ProgressIndicator((page*32U), Ipl_IpfData.StringSize);
224  }
225  if (IPL_RES_OK == res)
226  {
227  /* Get CRC */
228  Ipl_ClrTel();
229  Ipl_IplData.Tel[0] = CMD_GETCRC;
230  Ipl_IplData.Tel[3] = 0x2U;
231  Ipl_IplData.TelLen = CMD_GETCRC_TXLEN;
232  res = Ipl_ExecInicCmd();
233  if(IPL_RES_OK == res)
234  {
235  if ( (Ipl_IplData.Tel[4] != 0x00U) || (Ipl_IplData.Tel[5] != 0x00U) )
236  {
237  res = IPL_RES_ERR_WRONG_CRC;
238  }
239  }
240  }
241  }
242  }
243  }
244  }
245  }
246  }
247  Ipl_ProgressIndicator(1U, 1U); /* Set Progress Indicator to 100 */
248  Ipl_Trace(Ipl_TraceTag(res), "OS81110_ProgConfigString returned 0x%02X", res);
249  return res;
250 }

254 uint8_t OS81110_ReadConfigStringVersion(uint32_t lData, uint8_t pData[])
255 {
256  uint8_t res;
257  Ipl_Trace(IPL_TRACETAG_INFO, "OS81110_ReadConfigStringVersion called");
258  /* Get CS Version */
259  Ipl_ClrTel();
260  Ipl_IplData.Tel[0] = CMD_LEG_GETCSINFO;
261  Ipl_IplData.Tel[3] = 0x1AU;
262  Ipl_IplData.TelLen = CMD_LEG_GETCSINFO_TXLEN;
263  res = Ipl_ExecInicCmd();
264  if (IPL_RES_OK == res)
265  {
266  Ipl_InicData.CfgsVersionValid = VERSION_VALID;
267  Ipl_InicData.CfgsCustMajorVersion = Ipl_IplData.Tel[27];
268  Ipl_InicData.CfgsCustMinorVersion = Ipl_IplData.Tel[28];
269  Ipl_InicData.CfgsCustReleaseVersion = Ipl_IplData.Tel[29];
270  Ipl_Trace(Ipl_TraceTag(res), "OS81110_ReadConfigStringVersion returned 0x%02X - V%u.%u.%u", res,
271  Ipl_InicData.CfgsCustMajorVersion, Ipl_InicData.CfgsCustMinorVersion, Ipl_InicData.CfgsCustReleaseVersion);
272  }
273  else
274  {
275  Ipl_Trace(Ipl_TraceTag(res), "OS81110_ReadConfigStringVersion returned 0x%02X", res);
276  }
277  return res;
278 }
