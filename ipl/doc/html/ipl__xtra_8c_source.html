47 uint8_t Ipl_WriteIOReg(uint8_t reg, uint16_t val, uint16_t mask)
48 {
49  uint8_t res = IPL_RES_ERR_NOT_SUPPORTED;
50  Ipl_Trace(IPL_TRACETAG_INFO, "Ipl_WriteIOReg called");
51  Ipl_ClrTel();
52  Ipl_IplData.Tel[0] = CMD_WRITEIOREG;
53  Ipl_IplData.Tel[1] = 0x00U;
54  Ipl_IplData.Tel[2] = reg;
55  Ipl_IplData.Tel[3] = 4U;
56  Ipl_IplData.Tel[4] = (val >> 8) & 0xFFU;
57  Ipl_IplData.Tel[5] = val & 0xFFU;
58  Ipl_IplData.Tel[6] = (mask >> 8) & 0xFFU;
59  Ipl_IplData.Tel[7] = mask & 0xFFU;
60  Ipl_IplData.TelLen = CMD_WRITEIOREG_TXLEN;
61  res = Ipl_ExecInicCmd();
62  Ipl_Trace(Ipl_TraceTag(res), "Ipl_WriteIOReg 0x%02X, Value: 0x%04X, Mask: 0x%04X returned 0x%02X", reg, val, mask, res);
63  return res;
64 }

73  Ipl_IplData.Tel[1] = 0x00U;
74  Ipl_IplData.Tel[2] = reg;
75  Ipl_IplData.Tel[3] = 1U;
76  Ipl_IplData.TelLen = CMD_READIOREG_TXLEN;
77  res = Ipl_ExecInicCmd();
78  val = (uint16_t) ( (Ipl_IplData.Tel[4] << 8) + Ipl_IplData.Tel[5] );
79  Ipl_Trace(Ipl_TraceTag(res), "Ipl_ReadIOReg 0x%02X returned 0x%02X, Value: 0x%04X", reg, res, val);
80  return res;
81 }
